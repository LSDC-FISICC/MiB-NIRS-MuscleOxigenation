4/**
5 * @brief Configure system clock to 64 MHz using PLL
6 * @details Complete PLL and RCC configuration sequence:
7 * 1. Set PLLMUL to 0x0E (×16) for HSI input
8 * 2. Configure Flash for 64 MHz operation (2 wait states)
9 * 3. Enable PLL oscillator and wait for lock
10 * 4. Set APB1 prescaler to 2 (32 MHz)
11 * 5. Switch SYSCLK source from HSI to PLL
12 * 6. Update SystemCoreClock variable
13 *
14 * ### Clock Calculation (STM32F303K8)
15 * - Input: HSI = 8 MHz
16 * - PLL Multiplier: 16 (PLLMUL = 0x0E)
17 * - SYSCLK = (8 MHz / 2) × 16 = 64 MHz
18 * - AHB Clock: 64 MHz (no prescaler)
19 * - APB1 Clock: 32 MHz (prescaler = 2)
20 * - APB2 Clock: 64 MHz (no prescaler)
21 *
22 * ### Register Operations
23 * - RCC_CFGR[21:18] = 0x0E (PLLMUL = 16)
24 * - FLASH_ACR[2:0] = 0x2 (Latency = 2 cycles for 64 MHz)
25 * - RCC_CR[24] = 1 (Enable PLL)
26 * - RCC_CFGR[1:0] = 0x2 (SW = PLL)
27 * - RCC_CFGR[10:8] = 0x4 (PPRE1 = divide by 2)
28 *
29 * @param None
30 * @return void
31 */
32void clk_config(void) {
33 // Set PLLMUL to 16: (8 MHz / 2) × 16 = 64 MHz system clock
34 RCC->CFGR |= 0xE << 18;
35 // Configure Flash for 2 wait states (required for 64 MHz operation)
36 FLASH->ACR |= 0x2;
37 // Enable PLL
38 RCC->CR |= RCC_CR_PLLON;
39 // Wait for PLL to lock
40 while (!(RCC->CR & RCC_CR_PLLRDY));
41 // Switch SYSCLK to PLL, set APB1 prescaler to 2 (32 MHz)
42 RCC->CFGR |= 0x402;
43 // Wait for system clock to switch to PLL
44 while (!(RCC->CFGR & RCC_CFGR_SWS_PLL));
45 // Update SystemCoreClock global variable
46 SystemCoreClockUpdate();
47}
