16/**
17 * @brief Initialize USART2 for configurable baud rate transmission
18 * @details Complete USART2 setup sequence:
19 * 1. Enable GPIOA and USART2 clocks
20 * 2. Configure PA2 (TX) and PA15 (RX) as AF7 (Alternate Function 7)
21 * 3. Configure BRR for desired baud rate
22 * 4. Enable transmitter, receiver, and receiver interrupt
23 *
24 * @param baud_rate - Desired baud rate (e.g., 460800 as used in this project)
25 * @return void
26 *
27 * @timing
28 * - Total init time: <1 ms
29 * - USART2 ready immediately after function returns
30 *
31 * @warning Call once at system startup, BEFORE any USART2_Send operations
32 * @see USART2_Send, USART2_putString
33 */
34void UART_Config(uint32_t baud_rate) {
35 // Enable GPIOA clock (for PA2 and PA15)
36 RCC->AHBENR |= RCC_AHBENR_GPIOAEN;
37 // Enable USART2 clock (APB1)
38 RCC->APB1ENR |= RCC_APB1ENR_USART2EN;
39
40 // Configure PA2 and PA15 as alternate function pins
41 GPIOA->MODER |= (0x02 << 4) | (0x2 << 30);
42 // Set PA2 alternate function to AF7 (USART2_TX)
43 GPIOA->AFR[0] |= (0x07 << 8);
44 // Set PA15 alternate function to AF7 (USART2_RX)
45 GPIOA->AFR[1] |= (0x07 << 28);
46
47 // Configure baud rate: BRR = (APB1_Clock / baud_rate)
48 // APB1 = 32 MHz (SYSCLK/2), so BRR = 32MHz / baud_rate
49 USART2->BRR = (uint32_t)((SystemCoreClock / 2) / baud_rate);
50 // Enable transmitter and receiver
51 USART2->CR1 |= USART_CR1_RE | USART_CR1_TE;
52 // Enable USART2
53 USART2->CR1 |= USART_CR1_UE;
54 // Enable receiver interrupt for future use
55 USART2->CR1 |= USART_CR1_RXNEIE;
56}

58/**
59 * @brief Send single character via USART2
60 * @details Blocks until previous transmission complete, then sends one byte
61 *
62 * @param c - Character byte to transmit
63 * @return void
64 *
65 * @timing
66 * - Per-byte latency: ~22 µs at 460800 baud (10 bits/byte: 8N1)
67 *
68 * @note Blocking function; waits for TC (transmission complete) flag
69 * @see UART_Config, USART2_putString
70 */
71void USART2_Send(uint8_t c) {
72 // Wait for previous transmission to complete (TC flag)
73 while (!(USART2->ISR & USART_ISR_TC));
74 // Load character into transmit data register
75 USART2->TDR = c;
76}

78/**
79 * @brief Send null-terminated string via USART2
80 * @details Transmits string character-by-character using USART2_Send()
81 *
82 * @param string - Pointer to null-terminated character string
83 * @return void
84 *
85 * @timing
86 * - Per-character latency: ~22 µs at 460800 baud (10 bits/byte: 8N1)
87 * - Example: 10-character string ≈ 220 µs total; typical CSV frame ~350 µs
88 *
89 * @data_format String must be null-terminated (\\0)
90 * @note Blocking function; waits for each character's transmission
91 * @see UART_Config, USART2_Send
92 */
93void USART2_putString(char *string) {
94 // Transmit each character until null terminator
95 while (*string) {
96 USART2_Send(*string);
97 string++;
98 }
