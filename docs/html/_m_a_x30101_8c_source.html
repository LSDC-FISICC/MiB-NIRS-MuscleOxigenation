16/**
17 * @brief Initialize MAX30101 in SpO2 mode (dual-LED: Red + IR)
18 * @details Configures sensor for blood oxygen (SpO2) measurement with low power consumption.
19 * - Mode: SpO2 (Red + IR LEDs)
20 * - Sample Rate: 50 Hz (SPO2_CONFIG = 0x01, bits [4:2] = 000)
21 * - ADC Resolution: 16-bit, 118 µs pulse width (SPO2_CONFIG bits [1:0] = 01)
22 * - ADC Range: 2048 nA full-scale (SPO2_CONFIG bits [6:5] = 00)
23 * - FIFO Configuration: No averaging, rollover enabled
24 * @param ledPower_red - Red LED drive current in milliamps (0.0 to 51.0 mA, ~0.2 mA steps)
25 * Converted to register value as: reg = (uint8_t)(mA / 0.2)
26 * Example: 1.6 mA → reg 0x08; 10 mA → reg 0x32
27 * @param ledPower_ir - IR LED drive current in milliamps (same range and conversion)
28 * @return void
29 * @note Suitable for battery-powered wearable applications.
30 * Call once during initialization before reading samples.
31 * @see MAX30101_InitMuscleOx
32 * @example
33 * MAX30101_InitSPO2Lite(1.6f, 1.6f); // 1.6 mA per LED, low power
34 * uint8_t samples = MAX30101_GetNumAvailableSamples();
35 */
36void MAX30101_InitSPO2Lite(float32_t ledPower_red, float32_t ledPower_ir) {
37 // Configure FIFO: no averaging, rollover enabled
38 I2C1_Write(SENSOR_ADDR, FIFO_CONFIG, 0x10);
39 // Select SpO2 mode (Red + IR)
40 I2C1_Write(SENSOR_ADDR, MODE_CONFIG, 0x03);
41 // SpO2 config: 2048 nA range, 50 Hz sample rate, 118 µs pulse width
42 I2C1_Write(SENSOR_ADDR, SPO2_CONFIG, 0x01);
43 // Reset FIFO read pointer
44 I2C1_Write(SENSOR_ADDR, FIFO_READPTR, 0x0);
45 // Reset FIFO write pointer
46 I2C1_Write(SENSOR_ADDR, FIFO_WRITPTR, 0x0);
47 // Set Red LED power
48 I2C1_Write(SENSOR_ADDR, LED1_PAMPLI, (uint8_t)(ledPower_red / 0.2f)); // Convert mA to register value (0.2 mA steps)
49 // Set IR LED power
50 I2C1_Write(SENSOR_ADDR, LED2_PAMPLI, (uint8_t)(ledPower_ir / 0.2f)); // Same LED power for IR
51}

53/**
54 * @brief Initialize MAX30101 for NIRS muscle oxygenation measurement (multi-LED: Red + IR + Green)
55 * @details Configures sensor for Near-Infrared Spectroscopy with 3 simultaneous LEDs.
56 * Optimal for non-invasive tissue oxygenation and perfusion monitoring.
57 * - Mode: Multi-LED (Red + IR + Green)
58 * - Sample Rate: 50 Hz (SPO2_CONFIG = 0x01, bits [4:2] = 000)
59 * - ADC Resolution: 16-bit, 118 µs pulse width (SPO2_CONFIG bits [1:0] = 01)
60 * - ADC Range: 2048 nA full-scale (SPO2_CONFIG bits [6:5] = 00)
61 * - FIFO Configuration: No averaging, rollover enabled
62 * @param ledPower_red - Red LED drive current in milliamps (0.0 to 51.0 mA, ~0.2 mA steps)
63 * @param ledPower_ir - IR LED drive current in milliamps (same range)
64 * @param ledPower_green - Green LED drive current in milliamps (same range)
65 * Converted to register value as: reg = (uint8_t)(mA / 0.2)
66 * @return void
67 * @note Three-LED configuration increases tissue penetration depth for muscle assessment.
68 * Same 50 Hz sample rate as SPO2Lite but with additional Green LED for NIRS analysis.
69 * @see MAX30101_InitSPO2Lite
70 * @example
71 * MAX30101_InitMuscleOx(10.0f, 10.0f, 10.0f); // 10 mA per LED
72 * uint8_t available = MAX30101_GetNumAvailableSamples();
73 * if (available > 0) MAX30101_ReadSingleCurrent(&sample);
74 */
75void MAX30101_InitMuscleOx(float32_t ledPower_red, float32_t ledPower_ir, float32_t ledPower_green) {
76 // Configure FIFO: no averaging, rollover enabled
77 I2C1_Write(SENSOR_ADDR, FIFO_CONFIG, 0x10);
78 // Select multi-LED mode (Red + IR + Green)
79 I2C1_Write(SENSOR_ADDR, MODE_CONFIG, 0x07);
80 // SpO2 config: 2048 nA range, 50 Hz sample rate, 118 µs pulse width
81 I2C1_Write(SENSOR_ADDR, SPO2_CONFIG, 0x01);
82 // Reset FIFO read pointer
83 I2C1_Write(SENSOR_ADDR, FIFO_READPTR, 0x0);
84 // Reset FIFO write pointer
85 I2C1_Write(SENSOR_ADDR, FIFO_WRITPTR, 0x0);
86 // Set Red LED power
87 I2C1_Write(SENSOR_ADDR, LED1_PAMPLI, (uint8_t)(ledPower_red / 0.2f));
88 // Set IR LED power
89 I2C1_Write(SENSOR_ADDR, LED2_PAMPLI, (uint8_t)(ledPower_ir / 0.2f));
90 // Set Green LED power
91 I2C1_Write(SENSOR_ADDR, LED3_PAMPLI, (uint8_t)(ledPower_green / 0.2f));
92}

94/**
95 * @brief Query FIFO status from MAX30101 sensor
96 * @details Reads FIFO write and read pointer registers to determine number of unread samples.
97 * Accounts for circular 32-sample FIFO with pointer wrap-around.
98 * @param None
99 * @return uint8_t Number of complete samples available (0 to 32)
100 * - Returns 0 if FIFO empty or pointers equal
101 * - Returns 1 to 32 for available samples
102 * @note Call before MAX30101_ReadSingleCurrentSpO2() or MAX30101_ReadSingleCurrent()
103 * to confirm new data is ready.
104 * @warning Multiple fast consecutive reads may show inconsistent results
105 * due to FIFO updates during pointer reads.
106 * @see MAX30101_ReadSingleCurrentSpO2, MAX30101_ReadSingleCurrent, MAX30101_UpdateReadPointer
107 * @example
108 * uint8_t count = MAX30101_GetNumAvailableSamples();
109 * if (count > 0) {
110 * MAX30101_ReadSingleCurrentSpO2(&sample);
111 * MAX30101_UpdateReadPointer(count);
112 * }
113 */
114uint8_t MAX30101_GetNumAvailableSamples(void) {
115 uint8_t write_ptr = 0;
116 uint8_t read_ptr = 0;
117 uint8_t num_samples = 0;
118
119 // Read FIFO write and read pointers from sensor
120 I2C1_Read(SENSOR_ADDR, FIFO_WRITPTR, &write_ptr, 1);
121 I2C1_Read(SENSOR_ADDR, FIFO_READPTR, &read_ptr, 1);
122
123 // Mask to 5 bits (FIFO pointers are 5-bit: 0-31)
124 write_ptr &= 0x1F;
125 read_ptr &= 0x1F;
126
127 // Calculate number of available samples (handles wrap-around)
128 if (write_ptr >= read_ptr) {
129 num_samples = write_ptr - read_ptr;
130 } else {
131 num_samples = (32 - read_ptr) + write_ptr;
132 }
133
134 return num_samples;
135}

137/**
138 * @brief Update the FIFO read pointer
139 * @details Advances the read pointer by a specified number of samples, wrapping around at 32.
140 * @param num_samples - [in] Number of samples to advance the read pointer
141 * @return void
142 */
143void MAX30101_UpdateReadPointer(uint8_t num_samples) {
144 uint8_t read_ptr = 0;
145 // Read current FIFO read pointer
146 I2C1_Read(SENSOR_ADDR, FIFO_READPTR, &read_ptr, 1);
147 // Advance pointer by num_samples with wrap-around at 32
148 read_ptr = (read_ptr + num_samples) % 32;
149 // Write updated pointer back to sensor
150 I2C1_Write(SENSOR_ADDR, FIFO_READPTR, read_ptr);
151}
