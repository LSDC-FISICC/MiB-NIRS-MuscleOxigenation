43/**
44 * @brief System initialization and main control loop
45 * @details Initializes all peripherals in sequence:
46 * 1. **Clock**: PLL to 64 MHz (HSI 8 MHz × 16)
47 * 2. **GPIO**: Status LED on PB3 (push-pull output)
48 * 3. **I2C1**: 400 kHz fast-mode on PB6 (SCL), PB7 (SDA)
49 * 4. **Sensor**: MAX30101 SpO2 Lite mode — Red + IR at 50 Hz, 1.6 mA each
50 * 5. **UART**: USART2 at 460800 baud (PA2=TX, PA15=RX)
51 * 6. **Timer**: SysTick at 50 Hz (20 ms period), enabling the acquisition ISR
52 *
53 * After initialization, the main loop waits for data_ready (set by SysTick ISR)
54 * and transmits each Red/IR sample pair over UART as a CSV string.
55 * All sensor acquisition runs in the ISR; main only handles transmission.
56 *
57 * @param None
58 * @return int - Never returns (infinite loop)
59 * @note Initialization order is critical: I2C must be configured before MAX30101,
60 * and UART before SysTick to avoid transmitting before the port is ready.
61 * @warning Enabling SysTick (last step) immediately arms the ISR. Any initialization
62 * that must complete before the first ISR fires should precede SysTick_Config().
63 * @execution
64 * - Blocking operations during init: PLL lock (~few µs), I2C configuration
65 * - Time to first ISR: ~20 ms after SysTick_Config()
66 * @see clk_config, LED_config, I2C1_Config, MAX30101_InitSPO2Lite, SysTick_Handler
67 * @example
68 * // After init, main loop outputs one line per sample at 50 Hz:
69 * // "1234.567,2345.678\r\n" (Red nA, IR nA)
70 */
71int main(void) {
72 // Configure system clock to 64 MHz via PLL
73 clk_config();
74 // Configure GPIO port B pin 3 as push-pull output for LED
75 LED_config();
76 // Configure I2C1 (400 kHz) for MAX30101 communication
77 I2C1_Config();
78 // Initialize MAX30101 for SpO2 measurement with medium LED power
79 MAX30101_InitSPO2Lite(1.6f,1.6f); // 1.6 mA LED current for low power operation (up to 51 mA max)
80 // Configure USART2 (PA2=TX, PA15=RX) at 460800 baud for data transmission
81 UART_Config(460800);
82 // Configure SysTick for 20 ms interrupts (SYSTICK_FREQ_HZ = 50 Hz)
83 SysTick_Config(SystemCoreClock / SYSTICK_FREQ_HZ);
84
85 // Main loop: real work happens in SysTick_Handler ISR
86 for (;;) {
87 if(data_ready) {
88 sprintf(tx_buffer, "%.3f,%.3f\r\n", MAX30101_SingleSampleCurrentSpO2.red, MAX30101_SingleSampleCurrentSpO2.ir);
89 USART2_putString(tx_buffer);
90 data_ready = 0; // Reset flag after transmission
91 }
92 }
93}

95/**
96 * @brief SysTick Timer Interrupt Service Routine (20 ms period)
97 * @details Core real-time data acquisition routine:
98 * 1. Queries MAX30101 FIFO for available samples
99 * 2. If samples available: reads one sample and converts to nanoamps
100 * 3. Advances FIFO read pointer and signals main loop via data_ready flag
101 * 4. Toggles status LED (visual heartbeat)
102 *
103 * This ISR runs non-preemptively (highest priority) every 20 milliseconds,
104 * synchronized with the MAX30101 output data rate (50 Hz). In steady state,
105 * exactly one sample is available per interrupt.
106 *
107 * @param None
108 * @return void
109 * @note ISR Context
110 * - Execution time: ~1–2 ms (I2C reads dominate; ~0.5 ms per transaction)
111 * - Called at SysTick interrupt (cannot nest itself)
112 * - All registers preserved; no clobbering of main loop state
113 *
114 * @data_output
115 * Upon samples available:
116 * - Populates MAX30101_SingleSampleCurrentSpO2 (Red, IR fields in nanoamps)
117 * - Sets data_ready = 1 to signal main loop
118 * - Data valid until next ISR fires (~20 ms window)
119 *
120 * @timing
121 * - ISR rate: 50 Hz (20 ms period), matching MAX30101 ODR of 50 Hz
122 * - Steady state: exactly 1 sample per interrupt
123 * - At startup: up to 2 samples may be present before the first ISR fires;
124 * only the most recent is read, the pointer advances past all pending samples
125 * - Sample age at read: 0–20 ms depending on arrival time within the period
126 *
127 * @warning
128 * - data_ready is declared volatile so the compiler does not cache it in a
129 * register across the ISR boundary. Without volatile, -O1 or higher can
130 * optimize away the flag check in the main loop (undefined behavior in C).
131 * - I2C blocking: If I2C bus is busy, ISR execution may extend by several ms
132 *
133 * @see MAX30101_GetNumAvailableSamples, MAX30101_ReadSingleCurrentSpO2, LED_Toggle
134 * @example
135 * // ISR fires every 20 ms (50 Hz), synchronized to sensor output
136 * // MAX30101_SingleSampleCurrentSpO2 updated with one fresh Red/IR nA pair
137 * // LED toggles each tick → 25 Hz blink (20 ms on, 20 ms off)
138 */
139
140void SysTick_Handler(void) {
141 uint8_t available_samples = MAX30101_GetNumAvailableSamples();
142 if (available_samples > 0) {
143 MAX30101_ReadSingleCurrentSpO2(&MAX30101_SingleSampleCurrentSpO2);
144 MAX30101_UpdateReadPointer(available_samples);
145 data_ready = 1; // Set flag for main loop to process new data
146 }
147 LED_Toggle();
148}
