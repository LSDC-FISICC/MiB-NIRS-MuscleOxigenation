13/**
14 * @brief Initialize GPIO port B, pin 3 as push-pull output
15 * @details Complete GPIO setup for LED control:
16 * 1. **Enable GPIOB peripheral clock** (AHB domain)
17 * 2. **Configure PB3 mode** as output (push-pull)
18 * 3. **Configure PB4 mode** as output (reserved for future use)
19 * 4. **Ensure push-pull type** (OTYPER[3]=0, OTYPER[4]=0)
20 * 5. Ready for ODR writes (LED_On, LED_Off, LED_Toggle)
21 *
22 * ### Register Operations
23 * - RCC->AHBENR |= RCC_AHBENR_GPIOBEN
24 * Bit Enable: Transitions GPIOB from OFF to ON power state
25 * Effect: GPIO registers become readable/writable
26 * Latency: ~3 AHB cycles (~50 ns @ 64 MHz)
27 *
28 * - GPIOB->MODER &= ~((3<<6)|(3<<8))
29 * Clear bits [7:6] (PB3 mode) and [9:8] (PB4 mode)
30 * Purpose: Reset both pin modes to input (00)
31 *
32 * - GPIOB->MODER |= (1<<6) | (1<<8)
33 * Set bits [6]=1 (PB3 mode[0]=1) and [8]=1 (PB4 mode[0]=1)
34 * Result: MODER[7:6] = 01 (PB3 = output), MODER[9:8] = 01 (PB4 = output)
35 * Note: PB4 is configured as output here; I2C1 uses PB6/PB7 (AF4), not PB4
36 *
37 * ### GPIO State After Config
38 * | Feature | PB3 | PB4 |
39 * |---------|-----|-----|
40 * | Mode | Output | Output (reserved) |
41 * | Type | Push-pull | Push-pull |
42 * | Speed | Medium | Medium |
43 * | Initial ODR | 0 (LOW) | 0 (LOW) |
44 * | Pull | None | None |
45 *
46 * @param None
47 * @return void
48 *
49 * @timing
50 * - Clock stabilization: ~3 AHB cycles
51 * - Register writes: ~1 cycle each (3 writes = 3 cycles)
52 * - Total: <500 ns
53 * - PB3 ready for output: immediately after function returns
54 *
55 * @critical_notes
56 * - Must be called AFTER clk_config() to ensure SYSCLK is stable at 64 MHz
57 * - I2C1_Config() must be called AFTER this function to properly configure PB4 as I2C alternate function
58 * - Initialization order: clk_config() → LED_config() → I2C1_Config()
59 * - Do not call LED_config() after I2C1_Config() as it overwrites PB4 alternate function configuration
60 *
61 * @side_effects
62 * - GPIOB clock permanently enabled (not disabled in low-power mode by default)
63 * - PB3 is now an output; cannot be used as input, ADC, or timer capture
64 * - PB3 starts LOW (LED off); no explicit LED_Off() call needed
65 * - Draws ~1 mA additional idle current (GPIOB always powered)
66 *
67 * @power_note
68 * - GPIOB active power: ~1 mA @ 3.3V (all 16 pins)
69 * - PB3 push-pull output (idle): <1 µA when HIGH, <1 µA when LOW
70 * - LED current: 10-20 mA typical (dominates GPIO current by ~100×)
71 *
72 * @see LED_On, LED_Off, LED_Toggle
73 */
74void LED_config(void) {
75 RCC->AHBENR |= RCC_AHBENR_GPIOBEN;
76 GPIOB->MODER &= ~((3<<6)|(3<<8));
77 GPIOB->MODER |= (1<<6) | (1<<8);
78}

80/**
81 * @brief Drive PB3 output HIGH (turn LED on)
82 * @details Sets GPIO output data register bit 3 to logical HIGH (3.3 V).
83 * Push-pull driver sources current for LED to illuminate.
84 *
85 * @param None
86 * @return void
87 * @timing ~30 ns latency
88 * @see LED_Off, LED_Toggle, LED_config
89 */
90void LED_On(void) {
91 GPIOB->ODR |= (1<<3);
92}

94/**
95 * @brief Drive PB3 output LOW (turn LED off)
96 * @details Clears GPIO output data register bit 3 to logical LOW (0 V).
97 * LED current stops and LED extinguishes.
98 *
99 * @param None
100 * @return void
101 * @timing ~30 ns latency
102 * @see LED_On, LED_Toggle, LED_config
103 */
104void LED_Off(void) {
105 GPIOB->ODR &= ~(1<<3);
106}

108/**
109 * @brief Toggle PB3 output state (if HIGH → LOW; else LOW → HIGH)
110 * @details Inverts LED state via XOR operation on GPIO output data register.
111 * Ideal for periodic blinking feedback in SysTick ISR.
112 *
113 * @param None
114 * @return void
115 * @timing ~30-50 ns latency
116 * @note Called from SysTick_Handler() every 20 ms → 25 Hz blink (20 ms on, 20 ms off)
117 * @see LED_On, LED_Off, LED_config
118 */
119void LED_Toggle(void) {
120 GPIOB->ODR ^= (1<<3);
121}
