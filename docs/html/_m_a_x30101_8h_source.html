3 * @brief MAX30101 Optical Biosensor Driver for NIRS Muscle Oxygenation
4 * @details Complete driver for Maxim Integrated MAX30101 optical sensor supporting:

53/**
54 * @struct MAX30101_Sample
55 * @brief Raw FIFO sample data directly read from sensor (6 bytes) in multi-LED mode
56 * @details Intermediate storage for raw 2-byte ADC values from each LED channel.
57 * Format: 3 channels × 2 bytes (MSB, LSB) per sample
58 * Total: 6 bytes per complete sample in multi-LED mode
59 * @note Use MAX30101_ReadFIFO() to populate this structure
60 * @see MAX30101_ReadFIFO, MAX30101_ConvertSampleToUint16
61 */
62typedef struct {
63 uint8_t red[2]; /**< Red LED ADC raw bytes (MSB, LSB) */
64 uint8_t ir[2]; /**< IR LED ADC raw bytes (MSB, LSB) */
65 uint8_t green[2]; /**< Green LED ADC raw bytes (MSB, LSB) */
66} MAX30101_Sample;

68/**
69 * @struct MAX30101_SampleData
70 * @brief 16-bit unsigned integer representation of ADC counts
71 * @details Intermediate format after combining 2 raw bytes per channel.
72 * Range: 0 to 65535 (16-bit ADC counts)
73 * Represents: Photodiode current as raw ADC values

75 * @see MAX30101_ConvertSampleToUint16, MAX30101_ReadFIFO_Current

80 uint16_t green; /**< Green LED 16-bit value as uint16_t */
81} MAX30101_SampleData;
